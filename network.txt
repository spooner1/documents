library(STRINGdb)
string_db <- STRINGdb$new( version="10", species=9606,score_threshold=0, input_directory="" )
data <- read.table("test.tsv",header=T,sep="\t")
data_mapped <- string_db$map(data, "gene", removeUnmappedRows = TRUE )
hits <- data_mapped$STRING_id
png("test.png")
string_db$plot_network(hits[1:100])
dev.off()

payload_id <- string_db$post_payload(data_mapped$STRING_id, colors=data_mapped$color)

pdf("test.pdf")
string_db$plot_network( hits, payload_id=payload_id )
dev.off()
